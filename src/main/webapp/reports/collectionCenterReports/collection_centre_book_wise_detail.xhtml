function hidePaginatorBeforePrint() { let paginators = document.querySelectorAll('.ui-paginator'); paginators.forEach(el => el.style.display = 'none'); let filters = document.querySelectorAll('.ui-column-filter, .ui-filter-row'); filters.forEach(el => el.style.display = 'none'); } function restorePaginatorAfterPrint() { let paginators = document.querySelectorAll('.ui-paginator'); paginators.forEach(el => el.style.display = ''); let filters = document.querySelectorAll('.ui-column-filter, .ui-filter-row'); filters.forEach(el => el.style.display = ''); } window.onafterprint = function () { restorePaginatorAfterPrint(); };
