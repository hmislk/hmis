function handlePrint() { console.log("Print button clicked"); const convertChartToImage = function (widgetVarName) { return new Promise((resolve, reject) => { const chartWidget = PF(widgetVarName); if (!chartWidget || !chartWidget.cfg || !chartWidget.cfg.chart) { console.warn("Chart not available:", widgetVarName); return resolve(false); } const canvas = chartWidget.cfg.chart.canvas; const img = document.createElement("img"); img.onload = () => { canvas.parentNode.replaceChild(img, canvas); resolve(true); }; img.onerror = reject; img.src = chartWidget.cfg.chart.toBase64Image(); img.style.width = canvas.style.width; img.style.height = canvas.style.height; }); }; Promise.all([ convertChartToImage('countChartWidget'), convertChartToImage('serviceChartWidget') ]).then(() => { setTimeout(() => { window.print(); }, 300); }).catch((e) => { console.error("Error during chart conversion", e); window.print(); }); } @media print { body * { visibility: hidden !important; } .print-panel, .print-panel * { visibility: visible !important; } .print-panel { position: absolute !important; left: 0; top: 0; width: 100%; } }
