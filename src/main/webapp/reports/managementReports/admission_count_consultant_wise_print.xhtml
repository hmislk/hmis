// Maps off-screen export widget → visible img element id var chartMappings = [ { wv: 'expPrintSpecialtyLineWV', imgId: 'visSpecialtyLineImg', hiddenId: 'printForm:specialtyLineChartImg' }, { wv: 'expPrintSpecialtyBarWV', imgId: 'visSpecialtyBarImg', hiddenId: 'printForm:specialtyBarChartImg' }, { wv: 'expPrintDoctorLineWV', imgId: 'visDoctorLineImg', hiddenId: 'printForm:doctorLineChartImg' }, { wv: 'expPrintDoctorBarWV', imgId: 'visDoctorBarImg', hiddenId: 'printForm:doctorBarChartImg' } ]; // Populate visible img elements from off-screen charts so p:printer can copy them function populateChartImages() { chartMappings.forEach(function(m) { try { var widget = PF(m.wv); if (widget && widget.chart) { var img = document.getElementById(m.imgId); if (img) { img.src = widget.chart.toBase64Image('image/png', 1); } } } catch(e) { console.warn('Chart image populate failed: ' + m.wv, e); } }); } // Set printed-at timestamp on page load window.addEventListener('load', function() { var span = document.getElementById('printedAtSpan'); if (span) { var now = new Date(); span.textContent = now.toLocaleDateString('en-GB', {day:'2-digit', month:'short', year:'numeric'}) + ' ' + now.toLocaleTimeString('en-GB', {hour:'2-digit', minute:'2-digit', second:'2-digit', hour12:false}); } setTimeout(populateChartImages, 500); }); // Called by the PDF button before form submit function captureChartsForPdf() { chartMappings.forEach(function(m) { try { var widget = PF(m.wv); if (widget && widget.chart) { var b64 = widget.chart.toBase64Image('image/png', 1); var input = document.getElementById(m.hiddenId); if (input) input.value = b64; var img = document.getElementById(m.imgId); if (img) img.src = b64; } } catch(e) { console.warn('Chart capture failed: ' + m.wv, e); } }); }
